⚠ Database Setup Required

Run this SQL in your Supabase SQL editor to add multi-sport support:

-- Run in Supabase SQL Editor
CREATE TABLE IF NOT EXISTS sports (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name text NOT NULL,
  emoji text DEFAULT '🏅',
  has_circus boolean DEFAULT false,
  sort_order int DEFAULT 0,
  created_at timestamptz DEFAULT now()
);

-- Add sport_id to all tables (safe to run multiple times)
ALTER TABLE players ADD COLUMN IF NOT EXISTS sport_id uuid REFERENCES sports(id);
ALTER TABLE practices ADD COLUMN IF NOT EXISTS sport_id uuid REFERENCES sports(id);
ALTER TABLE attendance ADD COLUMN IF NOT EXISTS sport_id uuid;
ALTER TABLE circus ADD COLUMN IF NOT EXISTS sport_id uuid;
ALTER TABLE circus_history ADD COLUMN IF NOT EXISTS sport_id uuid;
ALTER TABLE strike_resets ADD COLUMN IF NOT EXISTS sport_id uuid;

-- Enable RLS policies (if needed)
ALTER TABLE sports ENABLE ROW LEVEL SECURITY;
CREATE POLICY IF NOT EXISTS "allow all" ON sports FOR ALL USING (true) WITH CHECK (true);
